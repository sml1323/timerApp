### useTopicSelect 훅 구현 가이드

```typescript
// src/features/session/hooks/useTopicSelect.ts
import { useState, useEffect, useCallback } from 'react';
import type { Topic } from '../../../domain/topics/topic';
import { loadTopics } from '../../topics/topic-service';

export function useTopicSelect() {
  const [topics, setTopics] = useState<Topic[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [selectedTopicId, setSelectedTopicId] = useState<string | null>(null);

  const fetchTopics = useCallback(async () => {
    setIsLoading(true);
    setError(null);
    const result = await loadTopics();
    if (result.ok) {
      setTopics(result.data);
    } else {
      setError(result.message);
    }
    setIsLoading(false);
  }, []);

  useEffect(() => { fetchTopics(); }, [fetchTopics]);

  const selectTopic = useCallback((id: string) => {
    setSelectedTopicId(id);
  }, []);

  const isReady = selectedTopicId !== null;

  return { topics, isLoading, error, selectedTopicId, selectTopic, isReady };
}
```

### TopicQuickSelectPanel ARIA 패턴

키보드 접근성은 `listbox` 패턴으로 구현한다 (WAI-ARIA Listbox Pattern):

```tsx
// 핵심 구조:
<div role="listbox" aria-label="주제 선택"
     aria-activedescendant={selectedTopicId ? `topic-option-${selectedTopicId}` : undefined}
     tabIndex={0}
     onKeyDown={handleKeyDown}
>
  {topics.map((topic) => (
    <div key={topic.id}
         id={`topic-option-${topic.id}`}
         role="option"
         aria-selected={topic.id === selectedTopicId}
         onClick={() => selectTopic(topic.id)}
         className={cn(styles.topicItem, topic.id === selectedTopicId && styles.selected)}
    >
      {topic.name}
    </div>
  ))}
</div>

// handleKeyDown:
// ArrowDown → 다음 항목 선택 (순환하지 않음, 마지막이면 유지)
// ArrowUp → 이전 항목 선택
// Enter → onSelectTopic(selectedTopicId) 또는 시작 CTA 트리거
// Home → 첫 번째 항목
// End → 마지막 항목
```

### HomeRoute 수정 가이드

```tsx
// src/app/routes/HomeRoute.tsx — 현재 플레이스홀더를 교체
import { TopicQuickSelectPanel } from '../../features/session/components/TopicQuickSelectPanel';
import { useTopicSelect } from '../../features/session/hooks/useTopicSelect';
import { Button } from '../../shared/ui/Button/Button';
import styles from './HomeRoute.module.css'; // [NEW]

export function HomeRoute() {
  const { topics, isLoading, error, selectedTopicId, selectTopic, isReady } = useTopicSelect();

  const handleStartSession = () => {
    // Epic 3에서 세션 시작 로직 연결 예정
    if (selectedTopicId) {
      console.info('[Story 2.4] 세션 시작 준비 완료:', selectedTopicId);
    }
  };

  return (
    <section className={styles.homePage}>
      <h1>홈</h1>
      {/* Study Status Summary Card → Story 5.2에서 구현 */}
      <section aria-label="주제 선택 및 세션 시작" className={styles.quickStartSection}>
        <h2>학습 시작</h2>
        <TopicQuickSelectPanel
          topics={topics}
          isLoading={isLoading}
          error={error}
          selectedTopicId={selectedTopicId}
          onSelectTopic={selectTopic}
        />
        <Button
          variant="primary"
          disabled={!isReady}
          onClick={handleStartSession}
        >
          학습 시작
        </Button>
      </section>
    </section>
  );
}
```

### 스타일 가이드

- 주제 리스트: `max-height` + `overflow-y: auto` (5개 이상일 때 스크롤)
- 선택된 항목: `background-color: var(--color-bg-selected)`, `border-left: 3px solid var(--color-primary)`
- 포커스 링: `outline: var(--focus-ring-width) solid var(--focus-ring-color)`
- 호버: `background-color: var(--color-bg-hover)`
- 빈 상태 캐릭터: `width: 80px`, `opacity: 0.7`, 중앙 정렬
- 간격: `gap: var(--spacing-md)`, 패딩: `var(--spacing-lg)`
- 폰트: 시스템 산세리프 (토큰에서 상속)

### 접근성 요구사항

- `role="listbox"` + `role="option"` — 스크린 리더가 선택 목록으로 인식
- `aria-activedescendant` — 현재 포커스된 항목 식별
- `aria-selected="true"` — 선택된 항목 상태 전달
- `tabIndex={0}` — 키보드 포커스 진입점
- 시작 CTA의 `disabled` 상태 → `aria-disabled` 자동 적용 (Button 컴포넌트 내장)
- 빈 상태 이미지 `alt=""` (장식용) + 옆 텍스트로 의미 전달
- 로딩 중 스피너 또는 `aria-busy="true"` 처리

### Anti-Patterns to Avoid

- ❌ `topic-repository.ts` 직접 import — 반드시 `topic-service.ts` 경유
- ❌ `sql-client.ts` 또는 Tauri 플러그인 직접 호출
- ❌ `features/topics/hooks/useTopics.ts`를 통째로 재사용 — 이 훅은 주제 CRUD용이므로 Quick Select 전용 `useTopicSelect` 별도 생성
- ❌ `inline style` 사용 — CSS Modules 전용
- ❌ `className` 문자열 직접 연결 — `cn()` 유틸 사용
- ❌ Tailwind CSS 또는 styled-components
- ❌ Zustand store 생성 — 이 스토리는 useState 기반 로컬 상태로 충분. Zustand는 Epic 3 세션 상태에서 도입
- ❌ `loading.svg`를 로딩 상태로 사용 — `loading.svg`는 세션 진행 중 전용
- ❌ 세션 시작 로직 구현 — Epic 3 범위. 시작 CTA는 `console.info`만 남기고 향후 연결
- ❌ 대시보드 카드 구현 — Story 5.2 범위
- ❌ 주간 목표 요약 구현 — Epic 4 범위. 현재는 단순 주제 이름만 표시
- ❌ 공통 Button 재구현 — `src/shared/ui/Button` 재사용
- ❌ EmptyState 공통 컴포넌트 생성 — 아직 2개 이상 feature에서 재사용 전이므로 인라인 구현
